define(["./MizarGlobe","./Utils/Stats","./Navigation/Navigation","./Layer/LayerFactory",'./Projection/ProjectionFactory',
        "./CoordinateSystem/CoordinateSystemFactory"],
    function (MizarGlobe,Stats,Navigation,LayerFactory,ProjectionFactory,
          CoordinateSystemFactory) {
      /**
        @name Mizar
        @class
        Mizar class
        This is the entry point for Mizar API
      */
      var Mizar = function () {
          Mizar.LayerFactory = new LayerFactory();
          Mizar.ProjectionFactory = new ProjectionFactory();
          Mizar.CoordinateSystemFactory = new CoordinateSystemFactory();
      };

      /**
       * Create and get Globe Object
       * @function createGlobe
       * @memberof Mizar.prototype
       *
       * @param {JSON} options
       * @return {MizarGlobe}
       * @example TODO
       */
      Mizar.prototype.createGlobe = function (options) {
        Mizar.MizarGlobe = new MizarGlobe(options);
        return Mizar.MizarGlobe;
      };

      /**
       * Create and get Stats Object
       * @function createStats
       * @memberof Mizar.prototype
       *
       * @param {String} view
       * @param {JSON} options
       * @return {Stats}
       * @example TODO
       */
      Mizar.prototype.createStats = function (view,options) {
        Mizar.Stats = new Stats( view, options );
        return Mizar.Stats;
      };

      /**
       * Create and get Navigation Object
       * @function createNavigation
       * @memberof Mizar.prototype
       *
       * @param {String} view
       * @return {Navigation}
       * @example TODO
       */
      Mizar.prototype.createNavigation = function (view) {
        Mizar.Navigation = new Navigation(view);
        return Mizar.Navigation;
      };

      /**
       * Return the Layer Factory
       * @function getLayerFactory
       * @memberof Mizar.prototype
       * @return {LayerFactory}
       * @example TODO
       */
      Mizar.prototype.getLayerFactory = function () {
        return Mizar.LayerFactory;
      };

      /**
       * Return the Projection Factory
       * @function getProjectionFactory
       * @memberof Mizar.prototype
       * @return {ProjectionFactory}
       * @example TODO
       */
      Mizar.prototype.getProjectionFactory = function () {
        return Mizar.ProjectionFactory;
      };

      /**
       * Return the Coordinate System Factory
       * @function getCoordinateSystemFactory
       * @memberof Mizar.prototype
       * @return {CoordinateSystemFactory}
       * @example TODO
       */

      Mizar.prototype.getCoordinateSystemFactory = function () {
        return Mizar.ProjectionFactory;
      };

      // Make object MIZAR available in caller web page
      window.Mizar = Mizar;

      return Mizar;
    });
