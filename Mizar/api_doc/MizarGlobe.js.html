define(["./Context/Globe", "./Renderer/GeoBound",
        "./Navigation/Navigation", "./Navigation/FlatNavigation", "./Renderer/FeatureStyle", "./Utils/Stats", "./Parser/KMLParser", "./Animation/PathAnimation", "./Utils/Numeric", "./Animation/SegmentedAnimation", // Others
        "./Renderer/PointRenderer", "./Renderer/LineStringRenderable", "./Renderer/PolygonRenderer", "./Renderer/LineRenderer"], // Renderers
    function (Globe, GeoBound,
              Navigation, FlatNavigation, FeatureStyle, Stats, KMLParser, PathAnimation, Numeric, SegmentedAnimation,
              PointRenderer,LineStringRenderable,PolygonRenderer,LineRenderer) {
                // TODO FL   Pourquoi y-a-t-il les renderers ??


        /**
          @name MizarGlobe
          @class
            Create an instance of a Mizar Globe object
        */

        var MizarGlobe = function (options) {
          this.Globe = new Globe(options);

          this.GeoBound = GeoBound;
          this.FeatureStyle = FeatureStyle;
          this.Navigation = Navigation;
          this.FlatNavigation = FlatNavigation;
          this.Stats = Stats;
          this.KMLParser = KMLParser;
          this.Numeric = Numeric;
          this.PathAnimation = PathAnimation;
          this.SegmentedAnimation = SegmentedAnimation;
        };

        MizarGlobe.prototype.getGlobe = function () {
          return this.Globe;
        };

        return MizarGlobe;


    });
